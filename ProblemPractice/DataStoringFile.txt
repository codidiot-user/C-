using System;
using System.Collections.Generic;
using System.Linq;
using System.IO;
class Program
{
    static void Main()
    {
    
        try{
        var path1 = @"D:\HCL\ProblemPractice\DataSet1.csv";
        if(!File.Exists(path1)) throw new FileNotFoundException("File1 is not in this location");
        var path2 = @"D:\HCL\ProblemPractice\DataSet2.csv";
        if(!File.Exists(path2)) throw new FileNotFoundException("File2 is not in this location");
        var location = @"D:\HCL\ProblemPractice\DataSor.txt";
        if(!File.Exists(location)) throw new FileNotFoundException("Data storing is not in this location");
        
        var csv1 = File.ReadAllLines(path1)
                   .Skip(1)
                   .Select(line => line.Split(','))
                   .Select(fields => new
                   {
                       Names = fields[0],
                       PNumber = int.Parse(fields[2])
                   });
        var csv2 = File.ReadAllLines(path2)
                   .Skip(1)
                   .Select(lines => lines.Split(','))
                   .Select(fields => new
                   {
                       Names = fields[0],
                       PNumber = int.Parse(fields[2])
                   });
        var ms = csv1.Join(csv2,
                           NamesOfStu1 => NamesOfStu1.Names,
                           NamesOfStu2 => NamesOfStu2.Names,
                           (NamesOfStu1, NamesOfStu2) => new
                           {
                               phn1 = NamesOfStu1.PNumber,
                               phn2 = NamesOfStu2.PNumber
                           });
        var dummy = "C-1        C-2\n";
        File.WriteAllLines();
        File.AppendAllText(location, dummy);
        foreach(var item in ms)
        {
            var app = $"{item.phn1}      {item.phn2}\n";
            File.AppendAllText(location, app);
        }
        Console.WriteLine("Data Stored successfully!");
        }
        
        catch (FileNotFoundException f1)
        {
            Console.WriteLine(f1.Message);
        }
                           

    }
}
